/**
 * Mengambil semua artikel dari sheet "Potret" dan mengembalikannya
 * sebagai array of objects. Hanya artikel dengan kolom Aktif = TRUE
 * yang akan ditampilkan.
 *
 * Dipanggil dari sisi klien (Index.html) saat halaman Potret dibuka.
 */
function getPotretContent() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName('Potret');

  // Buat sheet default jika belum ada
  if (!sheet) {
    sheet = createPotretSheet(ss);
  }

  const data = sheet.getDataRange().getValues();
  const result = [];

  // Mulai dari baris ke-2 (skip header)
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    const id        = row[0] ? row[0].toString().trim() : '';
    const judul     = row[1] ? row[1].toString().trim() : '';
    const kategori  = row[2] ? row[2].toString().trim() : 'Panduan';
    const penulis   = row[3] ? row[3].toString().trim() : '';
    const tanggal   = row[4] ? Utilities.formatDate(new Date(row[4]), "Asia/Jakarta", "yyyy-MM-dd") : '';
    const ringkasan = row[5] ? row[5].toString().trim() : '';
    const isi       = row[6] ? row[6].toString().trim() : '';
    const referensi = row[7] ? row[7].toString().trim() : '';
    const aktif     = row[8]; // boolean TRUE/FALSE

    // Lewati baris kosong atau yang tidak aktif
    if (!id || !judul) continue;
    if (aktif === false || aktif === 'FALSE' || aktif === 0) continue;

    result.push({ id, judul, kategori, penulis, tanggal, ringkasan, isi, referensi });
  }

  return result;
}
